#[derive(Debug)]
struct AboutMe {
    pronouns: Vec<String>,
    languages: Vec<String>,
    hobbies: Vec<String>,
    fun_fact: Vec<String>,
}

impl AboutMe {
    fn new(
        pronouns: Vec<String>,
        languages: Vec<String>,
        hobbies: Vec<String>,
        fun_fact: Vec<String>,
    ) -> Self {
        AboutMe {
            pronouns,
            languages,
            hobbies,
            fun_fact,
        }
    }
}

fn main() {
    let carlo = AboutMe::new(
        vec!["He".to_owned(), "Him".to_owned()],
        vec![
            "C".to_owned(),
            "C#".to_owned(),
            "Java".to_owned(),
            "Rust".to_owned(),
            "Scala".to_owned(),
        ],
        vec![
            "reading".to_owned(),
            "trekking for dummies".to_owned(),
            "model railways".to_owned(),
        ],
        vec![],
    );

    println!("{:#?}", carlo);
}
